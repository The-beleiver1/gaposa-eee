// ── SINGLE FACE SCAN with APPROVAL + IDENTITY MISMATCH ───────── let _pendingScan = null; // { u, key, conf, dist } const MATCH_THRESHOLD = 0.45; const FLAG_THRESHOLD = 0.60; async function doScan(){ if(!stream){setS('Start camera first.','err');return;} if(!loaded){setS('Models loading...','warn');return;} if(!activeCourse){setS('Select a course first — tap the course banner.','warn');openCourseModal();return;} const btn=document.getElementById('bscan'); btn.disabled=true;btn.textContent='⏳ Scanning...'; document.getElementById('bmulti').disabled=true; setS('Scanning...','info'); document.getElementById('confwrap').classList.remove('show'); document.getElementById('multi-results').style.display='none'; document.getElementById('scan-approval').style.display='none'; document.getElementById('mismatch-banner').style.display='none'; const vid=document.getElementById('vid');sizeOC(); const c=document.getElementById('oc'),ctx=c.getContext('2d');ctx.clearRect(0,0,c.width,c.height); try{ const det=await faceapi.detectSingleFace(vid,new faceapi.TinyFaceDetectorOptions({inputSize:320,scoreThreshold:.5})).withFaceLandmarks(true).withFaceDescriptor(); if(!det){setS('No face detected. Position face clearly in frame.','warn');addLog('warn','No face');resetScanBtn();return;} const dims=faceapi.matchDimensions(c,{width:c.width,height:c.height},true); const res=faceapi.resizeResults(det,dims); if(!matcher||!students.length){setS('No students enrolled. Go to Enroll tab.','err');drawB(ctx,res.detection.box,'No DB','#FF3B5C');resetScanBtn();return;} // Use euclideanDistance for strict matching let bestDist = Infinity, bestStudent = null; for(const u of students){ if(!u.descriptors||!u.descriptors.length) continue; for(const d of u.descriptors){ const dist = faceapi.euclideanDistance(det.descriptor, d); if(dist < bestDist){ bestDist = dist; bestStudent = u; } } } const conf = Math.round((1 - bestDist) * 100); document.getElementById('confwrap').classList.add('show'); document.getElementById('cpct').textContent=conf+'%'; document.getElementById('cfill').style.width=conf+'%'; // ── IDENTITY MISMATCH FLAG ─────────────────────────────────── if(bestDist > FLAG_THRESHOLD || !bestStudent){ fails++; drawB(ctx,res.detection.box,'Unknown','#FF3B5C'); setS('Face not recognised. Enrol this student first.','err'); addLog('err','No match — dist:'+bestDist.toFixed(3)); // Log security breach for extremely poor matches if(bestDist > FLAG_THRESHOLD && bestStudent){ _triggerMismatchAlert(bestStudent, bestDist); } openM('me');resetScanBtn();return; } // ── STRICT THRESHOLD CHECK ─────────────────────────────────── if(bestDist > MATCH_THRESHOLD){ fails++; // Show mismatch if distance is between 0.45 and 0.6 _triggerMismatchAlert(bestStudent, bestDist); drawB(ctx,res.detection.box,'Uncertain','#FF9500'); resetScanBtn();return; } const u = bestStudent; if(!isStudentForActiveCourse(u)){ fails++; drawB(ctx,res.detection.box,esc(u.name.split(' ')[0])+' Wrong Level','#FF9500'); setS('Warning: '+esc(u.name)+' is '+u.level+' — not enrolled in this '+activeCourse.level+' course.','warn'); addLog('warn','Level mismatch: '+u.name);resetScanBtn();return; } if(sessionMarked.has(u.matric)){ setS(esc(u.name.split(' ')[0])+' already marked for Week '+currentWeek+'.','warn'); drawB(ctx,res.detection.box,esc(u.name.split(' ')[0])+' done','#FF9500'); addLog('warn','Duplicate: '+u.name);resetScanBtn();return; } // ── SHOW APPROVAL UI ───────────────────────────────────────── drawB(ctx,res.detection.box,esc(u.name.split(' ')[0]),'#1366F5'); _pendingScan = { u, key: getScanKey(), conf, dist: bestDist }; document.getElementById('approval-name').textContent = u.name; document.getElementById('approval-meta').textContent = u.matric + ' · ' + u.level + (u.option?' · '+u.option:'') + ' · ' + conf + '% match'; document.getElementById('scan-approval').style.display = 'block'; setS('Verify identity below — ACCEPT or REJECT.','info'); btn.disabled=false; btn.innerHTML=' Scan 1 Face'; document.getElementById('bmulti').disabled=false; return; // don't call resetScanBtn — approval flow handles it }catch(e){setS('Error: '+e.message,'err');addLog('err',e.message);} resetScanBtn(); } async function approveAttendance(){ if(!_pendingScan) return; const {u, key} = _pendingScan; document.getElementById('scan-approval').style.display='none'; await markStudentPresent(u, key); const c=document.getElementById('oc'),ctx=c.getContext('2d'); drawB(ctx,{x:0,y:0,width:c.width,height:c.height},esc(u.name.split(' ')[0]),'#00C06A'); setS(esc(u.name)+' — Week '+currentWeek+' — PRESENT!','ok'); document.getElementById('m-wk').textContent=currentWeek; document.getElementById('mname').textContent=u.name; document.getElementById('mdept').textContent=u.level+(u.option?' · '+u.option:' · General')+' · '+activeCourse.code; document.getElementById('mtime').textContent='Marked at '+new Date().toTimeString().slice(0,8)+' on '+new Date().toLocaleDateString('en-GB'); openM('ms');renderWeeklyGrid(); _pendingScan = null; speak('Attendance accepted for ' + u.name.split(' ')[0]); } function rejectAttendance(){ if(!_pendingScan) return; const name = _pendingScan.u.name.split(' ')[0]; document.getElementById('scan-approval').style.display='none'; const c=document.getElementById('oc'),ctx=c.getContext('2d'); ctx.clearRect(0,0,c.width,c.height); setS('Scan rejected for '+esc(name)+'. Camera reset for next student.','warn'); addLog('warn','Scan REJECTED by admin: '+_pendingScan.u.name); _pendingScan = null; speak('Scan rejected. Ready for next student.'); } async function _triggerMismatchAlert(u, dist){ const banner = document.getElementById('mismatch-banner'); const detail = document.getElementById('mismatch-detail'); if(banner && detail){ detail.textContent = 'Current face does NOT match Matric No: ' + u.matric + ' (' + u.name + '). Distance: ' + dist.toFixed(3); banner.style.display = 'block'; setTimeout(() => { if(banner) banner.style.display='none'; }, 6000); } setS('⚡ IDENTITY MISMATCH — Matric: '+u.matric+' | Dist: '+dist.toFixed(3),'err'); addLog('err','IDENTITY MISMATCH: '+u.matric+' dist:'+dist.toFixed(3)); speak('Verification failed. Unauthorised user detected.'); // Log to security_breach_logs try{ await dbFB.collection('security_breach_logs').add({ timestamp: firebase.firestore.FieldValue.serverTimestamp(), matricUsed: u.matric, studentName: u.name, distance: dist, courseCode: activeCourse ? activeCourse.code : 'unknown', week: currentWeek, flaggedBy: currentUserName || 'system', type: 'identity_mismatch' }); }catch(e){ addLog('warn','Security log write failed: '+e.message); } }
